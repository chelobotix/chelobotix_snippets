Will_paginate

# controller
Post.paginate(page: params[:page], per_page: 30)

# view
<div class="flickr_pagination">
    <div class="page_info">
      <%= page_entries_info @posts %>
    </div>
    <%= will_paginate @posts, :container => false %>
</div>
