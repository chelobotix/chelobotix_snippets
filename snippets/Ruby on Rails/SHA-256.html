SHA-256

# Flow
[Tu servidor] <--Webhook--> [EazyPost]
1. Se comparte el Secret con EazyPost y tu servidor
2. Evento ocurre en EazyPost
3. EazyPost genera firma usando secret + payload
4. EazyPost envía payload + firma al webhook
5. Tu servidor recalcula firma usando payload + secret
6. Comparas ambas firmas
7. Si coinciden -> procesas evento
   Si no -> ignoras

# APP 1
require 'openssl'

message = 'estoesuntest'
secret = 'mi_secreto'
hmac = OpenSSL::HMAC.hexdigest('SHA256', secret, message)

# hmac ahora contiene el hash que enviarás a la otra app
puts hmac



# APP 2
received_message = 'estoesuntest'  # Este es el mensaje que recibes de la primera app
received_hmac = 'hmac_recibido'  # Este es el HMAC que recibes junto con el mensaje
secret = 'mi_secreto'

calculated_hmac = OpenSSL::HMAC.hexdigest('SHA256', secret, received_message)

# Compare
if ActiveSupport::SecurityUtils.secure_compare(calculated_hmac, received_hmac)
  puts 'El mensaje es auténtico'
else
  puts 'El mensaje ha sido alterado o el HMAC no es válido'
end
