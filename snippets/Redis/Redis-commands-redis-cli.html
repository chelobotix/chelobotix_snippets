def self.scan_pattern(pattern, count = 10)
  cursor = "0"
  keys = []

  loop do
    cursor, found_keys = @redis.scan(cursor, match: "#{pattern}*", count: count)
    keys += found_keys
    break if cursor == '0'
  end

  keys
end
